func CanUse(x)
 return item.GetCooldown(x) <= 0
 ^& item.CanActivate()

func E(L,R)
 equipR @R@
 equipL @L@

func BladeAt(i) 
 return screen.i = i & Blade()

var wait = 0
func Blade()
 ?item.GetCooldown(＂blade＂) <= 0
  wait++
  screen.Next()
  ?wait > 18
   equip blade
   activate R
   screen.ResetOffset()
   wait = 0
  return true
 return false

func InstantOne(h,y,v)
 ?(CanUse(y))
  ?h = l | h = L
   equipL @v@
   activate L
  :?h = r | h = R

   activate R
  return true
